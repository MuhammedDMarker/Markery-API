/**
 * Create a wave attack around the NPC
 * @param {object} NPC 
 */
function createDamageWave(npc) {
    var waveRadius = 14;       // How far the damage wave reaches (in blocks)
    var waveDamage = 70;       // How much damage the wave does (2 damage = 1 full heart)
    var knockUp = 0.5;        // How high players are knocked into the air by the wave
    var particleName = "minecraft:cloud"; // The particle used for the wave

    var world = npc.world;
    var pos = npc.getPos();

    // A. Play a sound effect so the impact feels heavy
    // playSoundAt(position, sound_name, volume, pitch)
    world.playSoundAt(pos, "minecraft:entity.generic.explode", 2, 1);

    // B. Create a simple circle of particles
    // We use a simple loop that goes from 0 to 360 degrees
    for (var i = 0; i < 360; i += 30) { 
        var rad = i * (Math.PI / 180); // Convert degrees to math format (radians)
        
        // Calculate the edge of the circle based on our waveRadius
        var x = npc.x + Math.cos(rad) * waveRadius;
        var z = npc.z + Math.sin(rad) * waveRadius;
        
        // Spawn the particle at the calculated edge
        // spawnParticle(particle, x, y, z, speedX, speedY, speedZ, speed, count)
        world.spawnParticle(particleName, x, npc.y + 0.1, z, 0, 0, 0, 0.1, 5);
    }

    // C. Find players to damage
    // getNearbyEntities(position, distance, type)
    // Type 1 = Players, Type 2 = NPCs, Type 5 = All Living Mobs
    var targets = world.getNearbyEntities(pos, waveRadius, 1);

    // D. Loop through everyone we found and hit them
    for (var j = 0; j < targets.length; j++) {
        var target = targets[j];
        
        // Make sure the NPC doesn't accidentally hit itself!
        if (target.getName() != npc.getName()) {
            target.damage(waveDamage);
            target.setMotionY(knockUp); // Tosses the player up slightly
        }
    }
}
