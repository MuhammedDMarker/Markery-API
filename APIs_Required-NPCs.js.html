var PathsImport = Java.type("java.nio.file.Paths");

var WrapperNpcAPI = Java.type("noppes.npcs.api.wrapper.WrapperNpcAPI"); //Custom NPC's NpcAPI wrapper class.
var API = WrapperNpcAPI.Instance(); //Getting the Wrapper of the NpcAPI

var base = API.getGlobalDir().getParent();
var libraryFolder = new FileImport(base, "libraries")
var MarkeryFolder = new FileImport(libraryFolder, "Markery");

var RequiredScriptsFolder = new FileImport(MarkeryFolder, "Required NPC Scripts"); if (!RequiredScriptsFolder.exists()) {RequiredScriptsFolder.mkdir()}
var EventScriptsFolder = new FileImport(MarkeryFolder, "Events"); if (!MarkeryFolder.exists()) {MarkeryFolder.mkdir()}
var EntitiesScriptsFolder = new FileImport(EventScriptsFolder, "Entities"); if (!EntitiesScriptsFolder.exists()) {EntitiesScriptsFolder.mkdir()}

var checkPlaceScript = new FileImport(RequiredScriptsFolder, "check-npc-place.js"); if (!checkPlaceScript.exists()) {print("File doesn't exist in Required-NPCs.js")}
var checkPlaceScriptContect = new java.lang.String(FilesImport.readString(PathsImport.get(checkPlaceScript.getAbsolutePath())));

/**
 * Insert The Check Place script to the NPC
 * @param {object} NPC 
 */
function insertCheckPlace(NPC) {
    var arrayList = [];

    var nbt = NPC.getEntityNbt();
    var list1 = API.stringToNbt("{}");
    list1.putString("Script", checkPlaceScriptContect);
    
    nbt.setBoolean("ScriptEnabled", true);
    arrayList.push(list1);

    nbt.setList("Scripts", arrayList);
    npc.setEntityNbt(nbt);
}
