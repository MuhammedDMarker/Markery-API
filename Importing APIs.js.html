/**
 * 
 * @param {object} MCPlayer 
 * @param {double} balance 
 */
function savePlayerBalanceMethod(mcPlayer, balance) {
    var economyCap = mcPlayer.getCapability(EconomyProvider.ECON_CAP).orElse(null);
    if (economyCap != null) {
        economyCap.setBalance(balance); 
    }
    ScoreboardHandler.updatePlayerScore(mcPlayer, balance);
    ScoreboardHandler.syncPlayerBalance(mcPlayer);
}
/**
 * 
 * @param {object} MCPlayer 
 * @returns {double}
 */
function getPlayerSavedBalanceMethod(mcPlayer) {
    var economyCap = mcPlayer.getCapability(EconomyProvider.ECON_CAP).orElse(null);
    if (economyCap != null) {
        return economyCap.getBalance();
    }
    return null;
}
